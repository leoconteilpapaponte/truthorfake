<script>

let questions = [

["Drinking water increases intelligence by 30%", false, "Fake: no evidence"],

["NASA hit an asteroid", true, "True: DART mission"],

["Man lived on pizza 3 months", false, "Fake: unrealistic"],

["EU USB-C law approved", true, "True"],

["Sharks walk on land", false, "Fake"],

["AI passed law exam", true, "True"],

["Chocolate doubles lifespan", false, "Fake"],

["Robot hotel in Japan exists", true, "True"],

["Aliens contacted Earth", false, "Fake"],

["Climate change increases extreme weather", true, "True"]

];


let i = 0;

let score = 0;

let time = 10;

let timer;


function start() {

document.getElementById("intro").style.display = "none";

document.getElementById("quiz").style.display = "block";

next();

}


function next() {

if(i >= questions.length){

end();

return;

}


document.getElementById("q").innerText = questions[i][0];

document.getElementById("result").innerText = "";

time = 10;

document.getElementById("timer").innerText = "Time: " + time;


clearInterval(timer);

timer = setInterval(()=>{

time--;

document.getElementById("timer").innerText = "Time: " + time;


if(time == 0){

clearInterval(timer);

wrong();

}

},1000);


updateBar();

}


function answer(a){

clearInterval(timer);


if(a == questions[i][1]){

score += 3;

document.getElementById("result").innerText = "Correct!";

}else{

wrong();

return;

}


document.getElementById("score").innerText = "Score: " + score;

i++;

setTimeout(next,1000);

}


function wrong(){

document.getElementById("result").innerText = "Wrong! " + questions[i][2];

i++;

setTimeout(next,1500);

}


function updateBar(){

let p = (i / questions.length) * 100;

document.getElementById("bar").style.width = p + "%";

}


function end(){

document.getElementById("quiz").style.display = "none";

document.getElementById("end").style.display = "block";


let msg = "";


if(score >= 27) msg = "Excellent!";

else if(score >= 18) msg = "Good job!";

else msg = "Be careful with fake news!";


document.getElementById("end").innerHTML =

"<h2>Finished</h2><p>Score: "+score+"/30</p><p>"+msg+"</p>";

}
